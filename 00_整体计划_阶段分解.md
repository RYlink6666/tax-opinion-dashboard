#### 2.2 Python规则引擎脚本构建（12月17-18日）

```python
# opinion_analyzer.py
import pandas as pd
import re
from datetime import datetime
from typing import Dict, List, Tuple

class OpinionAnalyzer:
    """跨境电商舆论自动分类引擎"""
    
    def __init__(self):
        """初始化关键词库"""
        # 模式识别库
        self.mode_keywords = {
            '0110': ['香港公司', '新加坡公司', '空壳', '实质管理地', 'ODI', '利润回不来', '分红'],
            '9610': ['9610', '备案', '核定征收', '物流', '三单对碰', '退运', '海外仓'],
            '9710': ['9710', 'B2B', '线上订单', '身份验证', '阿里国际站', '速卖通'],
            '9810': ['9810', '海外仓', '离境退税', '报关价格', '库存核销', '多平台混合'],
            '1039': ['1039', '市场采购', '外综服', '义乌', '小商户', '拼箱'],
            'Temu': ['Temu', '全托管', '内销', '平台定价'],
        }
        
        # 风险识别库
        self.risk_keywords = {
            '香港空壳': ['空壳', '虚拟', '实质管理地', '0申报', '无人员'],
            '备案难题': ['备案', '困难', '跑了三个月', '物流不懂', '手续复杂'],
            '库存核销': ['库存', '核销', '多平台混合', '财务算不清'],
            '数据不符': ['数据对不上', '口径不一致', '增值税所得税'],
            '恶意拆分': ['拆分', '几百个小主体', '超过500万'],
            '规模困境': ['超500万', '税负爆表', '利润不够交税'],
            '补税压力': ['补税', '滞纳金', '被查', '处罚'],
            '信息不透明': ['怎么办', '不知道', '不理解'],
        }
        
        # 情感词库
        self.positive_words = ['支持', '点赞', '认可', '感谢', '相信', '解决', '方案']
        self.negative_words = ['怎么办', '困惑', '焦虑', '风险', '被罚', '无处可逃', '崩溃']
        
        # 行为倾向词库
        self.behavior_keywords = {
            '补税': ['补税', '主动补税', '预留资金'],
            '切换': ['切换', '改用', '不走...走', '考虑改'],
            '求助': ['怎么办', '求助', '咨询', '找顾问'],
            '观望': ['等等看', '等政策', '再看看'],
        }
        
        # 参与者身份库
        self.actor_keywords = {
            '卖家': ['我的店', '我的货', '我在', '我们团队', '我的企业'],
            '顾问': ['作为顾问', '建议', '规范做法', '应该这样', '不合规'],
            '初学者': ['刚进入', '新手', '第一次', '小白'],
            '大卖': ['几个亿', '几百个店铺', '多平台', '规模大'],
            '小商户': ['小规模', '个体户', '不到500万'],
        }
    
    def extract_mode(self, text: str) -> str:
        """识别税收模式"""
        for mode, keywords in self.mode_keywords.items():
            for kw in keywords:
                if kw in text:
                    return mode
        return '无关'
    
    def extract_risks(self, text: str) -> List[str]:
        """识别风险类型"""
        risks = []
        for risk, keywords in self.risk_keywords.items():
            count = sum(text.count(kw) for kw in keywords)
            if count > 0:
                risks.append(risk)
        return risks if risks else ['无明确风险']
    
    def extract_sentiment(self, text: str) -> str:
        """识别情感"""
        pos_count = sum(text.count(w) for w in self.positive_words)
        neg_count = sum(text.count(w) for w in self.negative_words)
        
        if pos_count > neg_count:
            return 'positive'
        elif neg_count > pos_count:
            return 'negative'
        else:
            return 'neutral'
    
    def extract_behavior(self, text: str) -> str:
        """识别行为倾向"""
        for behavior, keywords in self.behavior_keywords.items():
            if any(kw in text for kw in keywords):
                return behavior
        return '无明确倾向'
    
    def extract_actor(self, text: str) -> str:
        """识别参与者身份"""
        for actor, keywords in self.actor_keywords.items():
            count = sum(text.count(kw) for kw in keywords)
            if count > 0:
                return actor
        return '未识别'
    
    def analyze(self, text: str, publish_date: str = None) -> Dict:
        """完整分析一条舆论"""
        return {
            'text': text,
            'sentiment': self.extract_sentiment(text),
            'mode': self.extract_mode(text),
            'risks': '|'.join(self.extract_risks(text)),
            'behavior': self.extract_behavior(text),
            'actor': self.extract_actor(text),
            'publish_month': publish_date[:7] if publish_date else None,
            'processed_date': datetime.now().isoformat(),
        }
    
    def batch_analyze(self, df: pd.DataFrame) -> pd.DataFrame:
        """批量处理CSV数据"""
        results = []
        
        for idx, row in df.iterrows():
            analysis = self.analyze(row['text'], row.get('publish_date'))
            analysis.update({
                'id': row.get('id'),
                'likes': row.get('likes'),
                'platform': row.get('platform'),
            })
            results.append(analysis)
            
            # 进度显示
            if (idx + 1) % 500 == 0:
                print(f"已处理 {idx + 1} 条")
        
        return pd.DataFrame(results)

# 使用示例
if __name__ == "__main__":
    # 1. 加载清洁数据
    df = pd.read_csv('weibo_clean_data.csv')
    print(f"总计 {len(df)} 条舆论")
    
    # 2. 初始化分析器
    analyzer = OpinionAnalyzer()
    
    # 3. 批量分析（只需1-2分钟）
    results_df = analyzer.batch_analyze(df)
    
    # 4. 保存结果
    results_df.to_csv('analysis_results.csv', index=False, encoding='utf-8')
    print("分析完成！结果已保存到 analysis_results.csv")
    
    # 5. 基础统计
    print("\n=== 快速统计 ===")
    print(f"情感分布：\n{results_df['sentiment'].value_counts()}")
    print(f"\n模式分布：\n{results_df['mode'].value_counts()}")
    print(f"\n行为倾向：\n{results_df['behavior'].value_counts()}")
```

具体计算：

【时间序列数据】
SELECT 
  publish_month,
  COUNT(*) as total_count,
  SUM(CASE WHEN sentiment='positive' THEN 1 ELSE 0 END) as positive_count,
  SUM(CASE WHEN sentiment='negative' THEN 1 ELSE 0 END) as negative_count,
  AVG(likes) as avg_likes
FROM analysis_results
GROUP BY publish_month
ORDER BY publish_month;

【模式分布】
SELECT 
  mode,
  COUNT(*) as count,
  COUNT(*)*100.0/SUM(COUNT(*)) OVER () as percentage,
  SUM(CASE WHEN sentiment='negative' THEN 1 ELSE 0 END) as negative_count
FROM analysis_results
WHERE mode != '无关'
GROUP BY mode
ORDER BY count DESC;

【风险排行】
SELECT 
  risk_category,
  COUNT(*) as mention_count,
  AVG(CASE WHEN sentiment='negative' THEN 1 ELSE 0 END) as negative_ratio,
  AVG(likes) as avg_engagement
FROM analysis_results
WHERE risk_category IS NOT NULL
GROUP BY risk_category
ORDER BY mention_count DESC;

【行为倾向时间序列】
SELECT 
  publish_month,
  behavior_intent,
  COUNT(*) as count
FROM analysis_results
GROUP BY publish_month, behavior_intent
ORDER BY publish_month, count DESC;
